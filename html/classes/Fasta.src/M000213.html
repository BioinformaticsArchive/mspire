  def load(io)
    current_prot = nil
    current_aaseq = nil
    @prots.clear
    io.each do |line|
      if line[0,1] == '>'
        current_prot = Prot.new
        @prots << current_prot
        current_prot.header = line.chomp
        current_aaseq = ''
        current_prot.aaseq = current_aaseq
      elsif (line =~ /[^ ]/) && (line.size > 1)
        current_aaseq << line.chomp 
      end
    end
    self
  end
