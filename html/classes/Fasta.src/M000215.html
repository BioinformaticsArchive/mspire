  def read_file(fn=nil)
    @filename = fn if fn
    first_char_re = /[A-Za-z*]/o
    obj = nil
    regex = /(\r\n)|\n/o
    fh = File.new(fn).binmode
    lines = fh.read.split(regex)
    fh.close
    first_char = nil
    lines.each do |line|
      if line =~ /[^ \n\r]/
        first_char = line[0,1]
        if first_char == '>'
          obj = Prot.new
          @prots << obj
          obj.header = line.dup
        elsif first_char =~ first_char_re
          obj.aaseq << line.chomp
        else
          raise "Line not in fasta format (between arrows): -->#{line}<--"
        end
      end
    end
    self
  end
