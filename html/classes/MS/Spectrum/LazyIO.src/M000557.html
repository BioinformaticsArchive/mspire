  def self.new(*args)
    if args.size == 5  # mzXMl
      MS::Spectrum::LazyIO::Peaks.new(*args)
    elsif args.size == 9   # other
      MS::Spectrum::LazyIO::Pair.new(*args)
    else
      raise RunTimeError, "must give 5 or 7 args for peak data and pair data respectively"
    end
  end
