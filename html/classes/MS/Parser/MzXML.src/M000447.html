  def self.fix_bad_scan_tags_from_io(io, out_io)
    regexp = /<\/scan>/ 
    end_scan_line = false

    io.each("\n") do |line|
      if end_scan_line && line =~ regexp
        # two end scan lines! # don't print to out_io

        end_scan_line = true
      elsif line =~ regexp
        out_io.print(line)
        end_scan_line = true
      else
        out_io.print(line)
        end_scan_line = false
      end
    end
  end
