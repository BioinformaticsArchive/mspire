  def self.by_scan_num(file, regex)
    arr = []
    File.open(file) do |fh|
      string = fh.read 
      matches = string.scan(@@scan_re)
      matches.each do |matched|
        if inner_match = regex.match(matched[1])
          index = matched[0].to_i
          arr[index] = yield(inner_match)
        end
      end
    end
    arr
  end
