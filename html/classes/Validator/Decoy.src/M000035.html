  def pephit_precision(peps, separate_peps=nil)
    if separate_peps
      calc_precision(peps.size, separate_peps.size, @decoy_to_target_ratio)
    else
      (norm, decoy) = partition(peps)
      calc_precision(norm.size, decoy.size, @decoy_to_target_ratio)
    end
  end
