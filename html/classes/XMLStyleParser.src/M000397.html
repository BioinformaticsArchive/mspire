  def self.available_xml_parsers
    require_parsers  
    parser_precedence = Parser_precedence.dup
    File_required.map do |k,v|
      unless $".any? {|req_file| req_file.match(v) }
        parser_precedence.delete(k) 
      end
    end
    parser_precedence
  end
