  def to_srg(srg_filename='bioworks.srg')
    File.open(srg_filename, 'w') do |v|
      @filenames.each do |srf_file|
        if File.exist? srf_file
          v.puts File.expand_path(srf_file)
        else
          v.puts srf_file
        end
      end
    end
    srg_filename
  end
