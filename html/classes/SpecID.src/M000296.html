  def self.new(file=nil, tp=nil)
    # this will need to be specialized for other groups later
    if file.is_a?(Array)
      # takes an array of srf filenames
      SRFGroup.new(file)
    elsif file 
      from_file(file, tp)
    else
      GenericSpecID.new      
    end
  end
