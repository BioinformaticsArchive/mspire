  def self.from_file(file, tp=nil)
    obj = nil
    unless tp
      tp = file_type(file)
    end
    obj = case tp
    when 'srf'
      #@hi_prob_best = false
      SRFGroup.new([file])
    when 'srg'
      #@hi_prob_best = false
      SRFGroup.new(file)
    when 'bioworks'
      #@hi_prob_best = false
      Bioworks.new(file)
    when 'protproph'
      #@hi_prob_best = true
      Proph::ProtSummary.new(file)
    when 'pepproph'
      Proph::PepSummary.new(file)
    when 'sqg'
      SQTGroup.new(file)
    when 'sqt'
      SQTGroup.new([file])
    else
      abort "UNRECOGNIZED file type for #{file}"
    end
    obj
  end
