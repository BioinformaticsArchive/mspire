  def from_handle(fh)
    Arrayed.each do |ky|
      self[ky] = []
    end
    pos = fh.pos 
    lines = []
    loop do 
      line = fh.gets
      if line && (line[0,1] == SQT::Header::Leader )
        lines << line
      else # reset the fh.pos and we're done
        fh.pos = pos
        break
      end
      pos = fh.pos 
    end
    from_lines(lines)
  end
