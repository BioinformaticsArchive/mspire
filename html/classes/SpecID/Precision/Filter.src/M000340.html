  def filter_and_validate_cmdline(args)
    (spec_id_obj, options, option_parser) = CmdlineParser.new.parse(args)
    if spec_id_obj == nil
      puts option_parser
      return
    end
    final_answer = SpecID::Precision::Filter.new.filter_and_validate(spec_id_obj, options)
  end
