  def protproph_script_info
    begin
      where = `which xinteract`
      reply = `#{where}`
    rescue Exception
      reply = ""
    end
    prophet = "TPP (version unknown)"  # put your version here if you can't get it dynamically
    if reply =~ /xinteract.*?\((TPP .*)\)/
      prophet = $1.dup
    end
    script_info { "ProteinProphet from: #{prophet}" }
  end
