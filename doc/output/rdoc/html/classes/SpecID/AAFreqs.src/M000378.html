  def actual_and_expected_number_containing_cysteines(pep_objs, cyst_freq)
    if pep_objs.is_a? Hash
      seqs = pep_objs.keys
    else
      seqs = pep_objs.map do |v|
        v.aaseq
      end
    end
    @aafreqs ||= {}
    @aafreqs[:C] = cyst_freq
    actual_and_expected_number(seqs, :C, 1)
  end
