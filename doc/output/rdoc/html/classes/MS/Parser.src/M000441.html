  def self.filetype_and_version(fh_or_filename)
    if fh_or_filename.is_a? IO
      fh = fh_or_filename
      found = nil
      # Test for RAW file:
      header = fh.read(18).unpack('@2axaxaxaxaxaxaxa').join
      if header == 'Finnigan'
        return [:raw, nil]
      end
      fh.rewind
      while (line = fh.gets)
        found = 
          case line
          when Mzxml_regexp
            mtch = $2.dup
            case mtch
            when /mzXML_([\d\.]+)/
              [:mzxml, $1.dup]
            when /MsXML/
              [:mzxml, '1.0']
            else
              abort "Cannot determine mzXML version!"
            end
          when Mzdata_regexp
            [:mzdata, $1.dup]
          end
        if found
          break
        end
      end
      fh.rewind
      found
    else
      File.open(fh_or_filename) do |fh|
        filetype_and_version(fh)
      end
    end
  end
