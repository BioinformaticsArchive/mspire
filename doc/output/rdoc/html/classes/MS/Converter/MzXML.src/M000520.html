  def self.find_mzxml_converter
    ENV['PATH'].split(/[:;]/).each do |path|
      Dir.chdir(path) do
        Potential_mzxml_converters.each do |pc|
          if File.exist? pc
            return File.join(path, pc)
          end
        end
      end
    end
    nil
  end
