  def initialize(file=nil)
    @scans_by_num = []
    if file
      ext = File.extname(file)
      case ext
      when '.mzXML'
        set_from_mzxml(file)
      when '.timeIndex'
        set_from_index_file(file)
      else
        raise ArgumentError, "#{self.class}.new doesn't recognize files of extension: #{ext}"
      end
    end
  end
