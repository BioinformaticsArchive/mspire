  def start_and_end_mz(mzxml_file)
    @fh = File.open(mzxml_file)
    ms_level = 0
    @line = ""
    while ms_level != 1
      ms_level = _el("msLevel").to_i
    end
    start_mz = _el("startMz").to_f
    end_mz = _el("endMz").to_f
    @fh.close
    [start_mz, end_mz]
  end
