  def endElement(name)
    if name == 'peaks'
      @get_peaks = false
      @spectrum << Spectrum.new(@current_peaks_string, @precision)
      @spectrum.context = @current_scan
    elsif name == 'precursorMz'
      @current_scan[5] = @current_scan[5].to_f
      @get_precursor_mz = false
    end
  end
