  def new_scan_from_hash(node)
    scan = MS::Scan.new  # array class creates one with 9 positions
    scan[0] = node['num'].to_i
    scan[1] = node['msLevel'].to_i
    scan[2] = node['retentionTime'][2...-1].to_f
    if x = node['startMz']
      scan[3] = x.to_f
      scan[4] = node['endMz'].to_f
    end
    scan
  end
