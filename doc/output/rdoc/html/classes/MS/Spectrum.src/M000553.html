  def base64_peaks_to_pairs(string, precision=32)
    data = base64_peaks_to_array(string, precision)
    ndata = []
    data.each_with_index do |dat,ind|
      if (ind % 2) == 0  # even
        arr = Array.new(2)
        arr[0] = dat
        ndata.push( arr )
      else
        ndata.last[1] = dat 
      end
    end
    ndata
  end
