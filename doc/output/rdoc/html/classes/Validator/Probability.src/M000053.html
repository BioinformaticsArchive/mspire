  def precision(objs)
    return 1.0 if objs.size == 0

    current_sum_one_minus_prob = 0.0

    # this should work!
    #objs.inject(0.0) {|sum,obj| sum + (1.0 - obj.probability) }

    objs.each do |obj|
      # SUM(1-probX)/#objs
      current_sum_one_minus_prob += 1.0 - obj.send(@prob_method)
    end
    prec = 1.0 - (current_sum_one_minus_prob / objs.size)
  end
