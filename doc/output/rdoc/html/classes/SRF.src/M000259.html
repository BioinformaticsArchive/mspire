  def read_scan_index(fh, num)
    ind_len = 24
    index = Array.new(num)
    unpack_string = 'III'
    st = ''
    ind_len.times do st << '0' end  ## create a 24 byte string to receive data
    num.times do |i|
      fh.read(ind_len, st)
      index[i] = st.unpack(unpack_string)
    end
    index
  end
