  def file_as_decoy_to_html(string)  
    lines = string.split("\n")
    #puts lines ?? is this supposed to be commented out?
    lines = lines.reject do |obj| obj =~ /\*{10}/ end
    lines.map! do |line| "#{line}<br/>" end
    "<div class=\"fppr\">
    <h3>Classification Analysis</h3>
    #{lines.join("\n")} 
    </div>"
  end
