  def from_line(line)
    line.chomp!
    ar = line.split(SQT::Delimiter)
    self[0] = ar[1].to_i
    self[1] = ar[2].to_i
    self[2] = ar[3].to_f
    self[3] = ar[4].to_f
    self[4] = ar[5].to_f
    self[5] = ar[6].to_f
    self[6] = ar[7].to_i
    self[7] = ar[8].to_i
    self[8] = ar[9]
    self[9] = ar[10]
    self[14] = SpecID::Pep.sequence_to_aaseq(self[8])
    self
  end
