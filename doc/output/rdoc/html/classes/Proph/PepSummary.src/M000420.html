    def get_version(file)
      answer = nil
      File.open(file) do |fh|
        8.times do
          line = fh.gets
          answer = 
            if line =~ Filetype_and_version_re_new
              $1.dup
            end
          break if answer
        end
      end
      raise(ArgumentError, "couldn't detect version in #{file}") unless answer
      answer
    end
