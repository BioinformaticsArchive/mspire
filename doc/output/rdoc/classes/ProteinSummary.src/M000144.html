  def create_precision_argv(file, opt)
    # include only those options specific
    new_argv = [file]
    if opt.prefix ; new_argv << '--prefix' end
    if opt.f ; new_argv << '-f' << opt.f end
    if opt.o ; new_argv << '-o' << opt.o end
    new_argv
  end
