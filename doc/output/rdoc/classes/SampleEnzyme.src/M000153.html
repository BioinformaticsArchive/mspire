  def initialize(name=nil)
    @num_missed_cleavages_regex = nil
    @sense = nil
    @cut = nil
    @no_cut = nil
    @name = name
    if @name
      # set the values if we recognize this name
      send("set_#{@name}".to_sym)
    end
    if block_given?
      yield(self)
    end
  end
