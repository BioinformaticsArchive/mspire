  def inspect
    atts = %w(num ms_level time start_mz end_mz) 
    display = atts.map do |att|
      if val = send(att.to_sym)
        "#{att}=#{val}"
      else
        nil
      end
    end
    display.compact!
    spec_display = 
      if spectrum
        spectrum.mzs.size
      else
        'nil'
      end
    "<MS::Scan:#{__id__} " + display.join(", ") + " precursor=#{precursor.inspect}" + " spectrum(size)=#{spec_display}" + " >"
  end
