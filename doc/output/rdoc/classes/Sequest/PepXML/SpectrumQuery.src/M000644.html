  def to_pepxml
    case Sequest::PepXML.pepxml_version
    when 18
      element_xml("spectrum_query", [:spectrum, :start_scan, :end_scan, :precursor_neutral_mass, :assumed_charge, :index]) do
        search_results.collect { |sr| sr.to_pepxml }.join
      end
    end
  end
