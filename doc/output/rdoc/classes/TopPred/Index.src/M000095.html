  def reference_to_key(reference)
    if reference
      ri = reference.index(' ')
      frst =
        if ri 
          reference[0...reference.index(' ')]
        else 
          reference
        end
      if frst
        frst.gsub(/[^0-9a-zA-Z]/,'_')
      else
        nil
      end
    else
      nil
    end
  end
