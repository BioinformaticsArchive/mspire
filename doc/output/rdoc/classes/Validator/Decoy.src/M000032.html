  def partition(peps)
    if @decoy_on_match 
      if @correct_wins
        peps.partition do |pep|
          !(pep.prots.all? {|prot| prot.reference.match(@constraint) })
        end
      else  # fp wins
        peps.partition do |pep|
          !(pep.prots.any? {|prot| prot.reference.match(@constraint) })
        end
      end
    else 
      if @correct_wins
        peps.partition do |pep|
          pep.prots.any? {|prot| prot.reference.match(@constraint) }
        end
      else
        peps.partition do |pep|
          pep.prots.all? {|prot| prot.reference.match(@constraint) }
        end
      end
    end
  end
