  def set_false_to_total_ratio(peps)
    if peps.size > 0
      expected = 0.0
      peps.each do |pep|
        expected += (1.0 - ((1.0 - @frequency)**pep.aaseq.size))
      end
      @false_to_total_ratio = expected / peps.size
    else
      @false_to_total_ratio = 1.0
    end
  end
