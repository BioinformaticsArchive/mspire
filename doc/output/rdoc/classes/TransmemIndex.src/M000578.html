  def self.filetype(file)
    tp = nil
    File.open(file) do |fh|
      while (line = fh.gets)
        case line
        when /SEQENCE/
          tp = :phobius 
          break
        when /    0  0 i/
          tp = :phobius  # if they don't have the headers, 
                         # this will pick it up if they have a 
                         # single prot without tm or signal peptide.
          break
        when /Algorithm specific parameters/
          tp = :toppred  # New text
          break
        when /<parameters>/
          tp = :toppred  # XML
          break
        end
      end
    end
    tp
  end
