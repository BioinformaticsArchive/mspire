  def area_under_curve(x,y)
    area = 0.0
    (0...(x.size-1)).each do |i|
      # determine which is larger 

      if y[i+1] >= y[i]
        y1 = y[i+1]; y0 = y[i] 
      else
        y0 = y[i+1]; y1 = y[i] 
      end
      area += (x[i+1]-x[i]).to_f * ( y0.to_f + (y1-y0).to_f/2 ) 
    end
    area
  end
