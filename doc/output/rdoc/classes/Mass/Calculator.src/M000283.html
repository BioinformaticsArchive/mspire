  def masses(aaseqs)
    aaseqs.map do |aaseq|
      sum = @final_add  # <- add in the initialization
      aaseq.split('').each do |let|
        if @mass_hash.key? let
          sum += @mass_hash[let]
        else
          abort "LETTER not found in mass_hash: #{let}"
        end
      end
      sum
    end
  end
